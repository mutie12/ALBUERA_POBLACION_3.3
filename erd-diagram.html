     %% ========== ENTITY DEFINITIONS ==========

     User {
         string _id PK
         string name
         string username
         string password
         string email
         string phone
         string barangay
         string barangayCode
         bool isBarangayVerified
         string department
         string badgeNumber
         string vehicleNumber
         string role "resident|admin|respondent"
         string status "pending|approved|rejected"
         string verificationCode
         date createdAt
     }

     Report {
         string _id PK
         ObjectId reporterId FK
         string reporterName
         string location
         string emergencyType
         string description
         string image
         string resolution
         string status "Pending|Responding|Resolved"
         ObjectId assignedTo.respondentId FK
         string assignedTo.respondentName
         date assignedTo.assignedAt
         date date
     }

     Notification {
         string _id PK
         ObjectId recipientId FK
         string recipientName
         ObjectId senderId FK
         string senderName
         string type "assignment|update|resolved|system|broadcast|emergency|resident-message|resident-emergency"
         string title
         string message
         ObjectId reportId FK
         string emergencyType
         string location
         string image
         string status "unread|read"
         date createdAt
     }

     News {
         string _id PK
         string title
         string content
         string category "announcement|safety-tip|update|alert"
         string priority "low|normal|high"
         ObjectId author.id FK
         string author.name
         string author.department
         bool isActive
         string hazardType
         string location
         string details
         bool smsSent
         date createdAt
         date updatedAt
     }

👤 User (Single Collection)

All users (Admins, Respondents, Residents) stored in one users collection. Differentiated by role field (discriminator). Contains common fields plus role-specific fields that may be null for certain roles.

👑 Admin Role

Users with role: "admin". Have elevated privileges for system management, user administration, and emergency oversight.

🚒 Respondent Role

Users with role: "respondent". Emergency responders assigned to incidents. May have barangay, station, badge number, and vehicle information.

🏠 Resident Role

Users with role: "resident". Citizens who can file emergency reports. Requires barangay verification before submitting reports.

🟣 Report

Emergency incidents filed by Residents. Contains location, type, description, image, resolution, status lifecycle. Assigned to Respondents by Admins.

🟠 Notification

Real-time alerts sent to Users (in-app + SMS via Twilio). Types: assignment, update, resolved, emergency, broadcast, etc.

⚫ News

Public announcements, safety tips, and alerts broadcast to all users. Managed by Admins. Supports categories and priority levels.
